\section{Results} \label{sec:results} 
\beq
X_{{\rm pf},i} = \frac{1}{2} \big[ X_{{\rm pf},i}^{(1)} + X_{{\rm pf},i}^{(2)}\big]
\eeq

The main metric we use to compare some observable $X$ of standard $N$-body simulations 
to the paired-fixed simulations is bias, which we estimate as follows: 
\beq
\beta_X = \frac{\overline{X}_{\rm std} - \overline{X}_{\rm pf}}{\sqrt{\sigma_{\rm std}^2 + \sigma_{\rm pf}^2}}
\eeq
